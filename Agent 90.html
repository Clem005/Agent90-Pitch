Understanding Linear Equations

Interactive Graph: y = 2x + 3

A linear equation creates a straight line when graphed. The equation y = 2x + 3 shows:

  • Slope (2): The line rises 2 units for every 1 unit right
  • Y-intercept (3): The line crosses the y-axis at point (0,3)

React Component Example

// Adaptive Learning Component
import React, { useState, useEffect } from 'react';
import { useAI } from './hooks/useAI';

const AdaptiveLessonPlayer = ({ studentId, lessonId }) => {
  const [content, setContent] = useState(null);
  const [learningStyle, setLearningStyle] = useState('visual');
  const { adaptContent, trackProgress } = useAI();

  useEffect(() => {
    const loadContent = async () => {
      const adaptedContent = await adaptContent(
        lessonId, 
        learningStyle, 
        studentId
      );
      setContent(adaptedContent);
    };
    loadContent();
  }, [lessonId, learningStyle]);

  const handleInteraction = (interaction) => {
    trackProgress(studentId, lessonId, interaction);
  };

  return (
    <div className="adaptive-lesson-player">
      {content && (
        <ContentRenderer 
          content={content}
          onInteraction={handleInteraction}
        />
      )}
    </div>
  );
};

Backend API Example

// AI Personalization Service
const express = require('express');
const { OpenAI } = require('openai');
const app = express();

app.post('/api/adapt-content', async (req, res) => {
  const { lessonId, studentProfile, learningStyle } = req.body;
  
  try {
    // Get base content
    const baseContent = await getLesson(lessonId);
    
    // AI adaptation prompt
    const prompt = `
      Adapt this lesson for a ${learningStyle} learner:
      ${baseContent}
      
      Student profile: ${JSON.stringify(studentProfile)}
      
      Return adapted content with appropriate format.
    `;
    
    const response = await openai.chat.completions.create({
      model: "gpt-4",
      messages: [{ role: "user", content: prompt }]
    });
    
    const adaptedContent = response.choices[0].message.content;
    
    // Log adaptation for analytics
    await logAdaptation(studentProfile.id, lessonId, adaptedContent);
    
    res.json({ 
      success: true, 
      content: adaptedContent 
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
